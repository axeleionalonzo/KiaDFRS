create table model(
model_id serial,
name varchar(50),

primary key(model_id)
);


create table report(
report_id serial,
model_id int,
report_date date,
model_name varchar(50),
client varchar(50),
address varchar(50),
contactNo varchar(50),
term varchar(50),
remarks varchar(250),

primary key(report_id),
foreign key(model_id) references model(model_id)
);
